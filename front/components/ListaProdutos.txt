// front/componentes/ListaProdutos

ExibirListaDeProdutos(produtos){
    if(produtos.tamanho == 0)
        ExibirMensagemListaVazia();
        return;
    para cada produto em produtos:
        ExibirCardProduto(produto);
}

ExibirCardProduto(produto){
    renderizar(produto.nome);
    renderizar(produto.categoria);
    renderizar(produto.preco);
    ExibirBadgeEstoque(produto.quantidade, produto.estoqueMinimo);
}

ExibirMensagemListaVazia(){
    renderizar("Nenhum produto encontrado.");
}
