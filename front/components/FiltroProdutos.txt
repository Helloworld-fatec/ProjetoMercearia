// front/componentes/FiltroProdutos

ExibirFiltroProdutos(onDigitar, onSelecionar){
    ExibirFiltroPorNome(onDigitar);
    ExibirFiltroPorCategoria(categorias, onSelecionar);
}

ExibirFiltroPorNome(onDigitar){
    input(tipo=texto, placeholder="Buscar por nome...", onChange=onDigitar);
}

ExibirFiltroPorCategoria(categorias, onSelecionar){
    select(
        placeholder="Filtrar por categoria",
        opcoes=categorias,
        onChange=onSelecionar
    );
}
