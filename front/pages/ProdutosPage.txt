// front/pages/ProdutosPage

estado {
    paginaAtual = 1;
    limite = 10;
    totalPaginas = 0;
    filtroNome = "";
    filtroCategoria = "";
}

CarregarPaginaAtual(){
    var resposta = BuscarProdutosPaginados(paginaAtual, limite);
    if(resposta == erro)
        ExibirMensagemDeErro("Não foi possível carregar os produtos.");
    totalPaginas = resposta.totalPaginas;
    ExibirListaDeProdutos(resposta.produtos);
    ExibirControlesDePaginacao(paginaAtual, totalPaginas);
}

AvancarPagina(){
    if(paginaAtual < totalPaginas)
        paginaAtual = paginaAtual + 1;
    CarregarPaginaAtual();
}

VoltarPagina(){
    if(paginaAtual > 1)
        paginaAtual = paginaAtual - 1;
    CarregarPaginaAtual();
}

AtualizarDados(){
    paginaAtual = 1;
    CarregarPaginaAtual();
}
