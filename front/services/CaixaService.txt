RegistrarVenda(venda){
    var resposta = ChamarAPI(RotaRegistrarVenda(venda));
    if(resposta == erro)
        return erro;
    return resposta.venda;
}

CancelarVenda(vendaId){
    var resposta = ChamarAPI(RotaCancelarVenda(vendaId));
    if(resposta == erro)
        return erro;
    return resposta.venda;
}
