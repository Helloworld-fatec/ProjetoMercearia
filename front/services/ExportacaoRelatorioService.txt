ExportarRelatorioVendas(dataInicio, dataFim, formato){
    var resposta = ChamarAPI(RotaExportarRelatorioVendas(dataInicio, dataFim, formato));
    if(resposta == erro)
        return erro;
    return resposta.arquivo;
}
