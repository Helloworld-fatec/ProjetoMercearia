ConsultarProdutoPorNome(telefone, termoBusca){
    var resposta = ChamarAPI(RotaConsultarProdutoPorNomeViaWhatsApp(telefone, termoBusca));
    se(resposta == erro)
        return erro;
    return resposta;
}
